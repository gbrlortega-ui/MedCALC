// função genérica para ir para o próximo da sequência
function irParaProximo(){
  const atual = location.pathname.split("/").pop() || "index.html";
  const proximo = proximoInstrumento[atual];
  if(proximo){
    window.location.href = proximo;
  }else{
    console.warn("Nenhum próximo instrumento configurado para:", atual);
  }
}
